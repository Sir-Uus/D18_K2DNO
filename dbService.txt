async TotalPengiriman() {
    try {
      const query = "SELECT COUNT(*) AS totalKirim FROM pengiriman";
      const result = await new Promise((resolve, reject) => {
        connection.query(query, (err, results) => {
          if (err) reject(err);
          resolve(results[0].totalKirim);
        });
      });
      return result;
    } catch (error) {
      console.error("Error in TotalPengiriman:", error);
      return null;
    }
  }
